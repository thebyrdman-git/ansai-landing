    üöÄ Quick Start Guide¶

    Run First Backup¶

    # Test connectivity and run backup
ansai-miraclemax-backup

# Expected output:
# - Backup started message
# - Component-by-component backup progress
# - Backup complete with size and duration
# - Location: ~/backups/miraclemax/<timestamp>

    Verify Backup¶

    # Check backup directory
ls -lh ~/backups/miraclemax/latest/

# Expected structure:
# - databases/     (encrypted n8n, homeassistant)
# - volumes/       (grafana, traefik-certs, etc.)
# - configs/       (infrastructure configs)
# - logs/          (container metadata)
# - MANIFEST.txt   (backup details)

    Test Restore (Dry-Run)¶

    # Test restore without making changes
ansai-miraclemax-restore --latest --dry-run

# Expected output:
# - Backup verification passed
# - Restore plan displayed
# - No actual changes made

    Monitor Backup Status¶

    # Check timer status
systemctl --user status ansai-miraclemax-backup.timer

# View backup logs
journalctl --user -u ansai-miraclemax-backup.service -n 50

# Check Prometheus metrics
curl http://localhost:9090/api/v1/query?query=miraclemax_backup_success

    üîß Configuration¶

    Backup Options¶

    # Standard backup (recommended - skip Prometheus)
ansai-miraclemax-backup

# Full backup (includes 50GB+ Prometheus data)
ansai-miraclemax-backup --full

# Skip encryption (NOT recommended for n8n)
ansai-miraclemax-backup --skip-encryption

# Verify backup integrity after creation
ansai-miraclemax-backup --verify

    Restore Options¶

    # Restore from latest backup
ansai-miraclemax-restore --latest

# Restore from specific backup
ansai-miraclemax-restore --from-backup 20251016_030000

# Restore only specific component
ansai-miraclemax-restore --latest --component homeassistant
ansai-miraclemax-restore --latest --component n8n
ansai-miraclemax-restore --latest --component grafana

# Dry-run (test without changes)
ansai-miraclemax-restore --latest --dry-run

# Force (skip confirmations - DANGEROUS)
ansai-miraclemax-restore --latest --force

    Modify Backup Schedule¶

    # Edit timer
systemctl --user edit ansai-miraclemax-backup.timer

# Change to 2:00 AM:
[Timer]
OnCalendar=
OnCalendar=*-*-* 02:00:00

# Reload systemd
systemctl --user daemon-reload
systemctl --user restart ansai-miraclemax-backup.timer
